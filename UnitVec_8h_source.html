00182 UnitVec<P,S>::perp() const {
00183     // Choose the coordinate axis which makes the largest angle
00184     // with this vector, that is, has the "least u" along it.
00185     const UnitVec<P,1> u(abs());    // reflect to first octant
00186     const int minAxis = u[0] <= u[1] ? (u[0] <= u[2] ? 0 : 2)
